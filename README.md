Operaciones Lógicas

AND - && - ^ - y
V V -> V
V F -> F
F V -> F
F F -> F

OR - || - v - o
V V -> V
V F -> V
F V -> V
F F -> F

¬ - Negación - !
V -> F
F -> V

Ejemplos:
A <- 4
B <- 10
C <- 5
(A >= B) && (B <= C) R// (F && F) -> F
(A <= B) && (B <= C) R// (V && F) -> F
(A <= B) || (B <= C) R// (V || F) -> V
(A >= B)! && (B <= C)! R// (V && V) -> V
